Примеры запросов
Готовые curl команды для тестирования
# Получить баланс
curl http://localhost:3000/api/balance \
  -H "X-Api-Key: YOUR_API_KEY"

# Ответ:
{
  "success": true,
  "data": {
    "username": "testuser",
    "balance": 1000,
    "api_key": "31d269e8..."
  }
}
# Отправить 50 звёзд
curl -X POST http://localhost:3000/api/send-stars \
  -H "Content-Type: application/json" \
  -H "X-Api-Key: YOUR_API_KEY" \
  -d '{"telegram_user_id": "123456789", "amount": 50}'

# Ответ:
{
  "success": true,
  "message": "Successfully sent 50 stars.",
  "data": {
    "amount": 50,
    "balance_remaining": 950
  }
}
# Отправить подарок
curl -X POST http://localhost:3000/api/send-gift \
  -H "Content-Type: application/json" \
  -H "X-Api-Key: YOUR_API_KEY" \
  -d '{"telegram_user_id": "123456789"}'

# Ответ:
{
  "success": true,
  "message": "Gift sent successfully.",
  "data": {
    "gift_id": "5170145012310081615",
    "stars": 15
  }
}
# История транзакций
curl "http://localhost:3000/api/transactions?page=1&limit=10" \
  -H "X-Api-Key: YOUR_API_KEY"

# Фильтр по типу
curl "http://localhost:3000/api/transactions?type=debit" \
  -H "X-Api-Key: YOUR_API_KEY"
